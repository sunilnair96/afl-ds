# Create a directory to save the data files if it doesn't exist
output_dir = 'data'
os.makedirs(output_dir, exist_ok=True)

for season in seasons:
    for team in teams:
        # Generate the URL based on the current season and team
        url = f"https://dfsaustralia.com/afl-fantasy-points/?team={team}&season={season}"

        # Load the page using Selenium
        driver.get(url)

        # Wait for the page to load completely (you might need to adjust the wait time)
        time.sleep(5)

        # Get the page source after it is fully loaded
        page_source = driver.page_source

        # Parse the page source with BeautifulSoup
        soup = BeautifulSoup(page_source, 'html.parser')

        # Find the table with id="fantasyPoints"
        table = soup.find('table', {'id': 'fantasyPoints'})

        # Check if the table is found
        if table:
            # Get the headers from the <thead>
            headers = [header.text.strip() for header in table.find_all('th')]

            # Get the <tbody> and rows within it
            tbody = table.find('tbody')
            rows = []
            if tbody:
                for row in tbody.find_all('tr'):
                    # Get the text from each cell in the row and strip extra whitespace
                    cells = [cell.text.strip() for cell in row.find_all('td')]
                    rows.append(cells)

                # Create a DataFrame with the collected headers and rows
                df = pd.DataFrame(rows, columns=headers)

                # Save the DataFrame to an Excel file in the specified format
                output_file = os.path.join(output_dir, f'afl_fantasy_points_{season}_{team}.xlsx')
                df.to_excel(output_file, index=False)

                print(f"Table for season {season} and team {team} scraped and saved as {output_file}.")
            else:
                print(f"No <tbody> found in the table for season {season} and team {team}.")
        else:
            print(f"Table with id 'fantasyPoints' not found for season {season} and team {team}.")

# Close the Selenium browser
driver.quit()
